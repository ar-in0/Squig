121
122 double getBitrate() const {
123 auto now = std::chrono::steady_clock::now();
124 auto duration =
125 std::chrono::duration_cast<std::chrono::seconds>(now - start_time)
126 .count();
127 if (duration == 0)
128 return 0;
129 return (bytes_sent * 8.0) / duration / 1000.0; // kbps
130 }
131
132 double getUptime() const {
133 auto now = std::chrono::steady_clock::now();
134 return std::chrono::duration_cast<std::chrono::seconds>(now - start_time)
135 .count();
136 }
